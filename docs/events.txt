    RETURN EVENT:MouseDown
  OF 'mouseup'
    RETURN EVENT:MouseUp
  OF 'mousein'
    RETURN EVENT:MouseIn
  OF 'mouseout'
    RETURN EVENT:MouseOut
  OF 'mousemove'
    RETURN EVENT:MouseMove
  OF 'alertkey'
    RETURN EVENT:AlertKey
  OF 'prealertkey'
    RETURN EVENT:PreAlertKey
  OF 'dragging'
    RETURN EVENT:Dragging
  OF 'drag'
    RETURN EVENT:Drag
  OF 'drop'
    RETURN EVENT:Drop
  OF 'scrolldrag'
    RETURN EVENT:ScrollDrag
  OF 'tabchanging'
    RETURN EVENT:TabChanging
  OF 'expanding'
    RETURN EVENT:Expanding
  OF 'contracting'
    RETURN EVENT:Contracting
  OF 'expanded'
    RETURN EVENT:Expanded
  OF 'contracted'
    RETURN EVENT:Contracted
  OF 'rejected'
    RETURN EVENT:Rejected
  OF 'droppingdown'
    RETURN EVENT:DroppingDown
  OF 'droppeddown'

    RETURN EVENT:ScrollTrack
  OF 'columnresize'
    RETURN EVENT:ColumnResize

  OF 'losefocus'

  OF 'timer'

  OF 'move'

    RETURN EVENT:Docked

    !'__C10__'
